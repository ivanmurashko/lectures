\makeatletter
\newcommand{\getnamereftext}[1]{%
  \@ifundefined{r@#1}{}{%
    \unexpanded\expandafter\expandafter\expandafter{%
      \expandafter\expandafter\expandafter\@thirdoffive\csname r@#1\endcsname
    }%
  }%
}


\usepackage{ifthen}
\newcommand{\myref}[2]{%
  {\ifthenelse{\equal{#2}{}}
    {
      {\nameref{#1} }%
    }%
    {#2}
  }
  {\expandafter\index{{\getnamereftext{#1}}}}
  (\autoref{#1})
}%
